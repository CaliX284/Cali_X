CREATE VIEW captain_stats AS
SELECT
    c.id,
    c.full_name,
    c.phone,
    c.subscription_price,
    c.notes,

    COALESCE(m.members_count, 0) AS members_count,

    COALESCE(t.total_revenue, 0) AS total_revenue,
    COALESCE(t.gym_share, 0) AS gym_share,
    COALESCE(t.captain_share, 0) AS captain_share,

    s.percentage AS gym_percentage,
    (100 - s.percentage) AS captain_percentage

FROM captains c
CROSS JOIN settings s
LEFT JOIN (
    SELECT
        captain_id,
        COUNT(*) AS members_count
    FROM members
    GROUP BY captain_id
) m ON m.captain_id = c.id
LEFT JOIN (
    SELECT
        captain_id,
        SUM(amount_paid) AS total_revenue,
        SUM(gym_amount) AS gym_share,
        SUM(captain_amount) AS captain_share
    FROM transactions
    WHERE direction = 'income'
      AND paid_at >= CURRENT_DATE - INTERVAL '30 days'
    GROUP BY captain_id
) t ON t.captain_id = c.id;

CREATE VIEW members_view AS
SELECT
    m.id,
    m.full_name,
    m.captain_name,
    m.captain_id,
    m.phone,
    m.age,
    m.notes,

    m.subscription_price,
    m.paid_amount,
    m.remaining_amount,

    m.subscription_start_date,
    m.subscription_end_date,
    m.last_payment_date,

    m.is_frozen,
    m.frozen_days,
    m.freeze_reason,

    -- عدد الأيام المتبقية
    (m.subscription_end_date - CURRENT_DATE) AS days_left,

    -- هل عليه فلوس
    (m.remaining_amount > 0) AS has_remaining,

    -- حالة الاشتراك
    CASE
        WHEN m.is_frozen THEN 'frozen'

        WHEN m.subscription_end_date <= CURRENT_DATE THEN 'expired'

        WHEN m.subscription_end_date <=
            CURRENT_DATE + (
                SELECT expiring_soon_days
                FROM settings
                LIMIT 1
            )
        THEN 'expiring'

        ELSE 'active'
    END AS subscription_status,

    -- النص العربي
    CASE
        WHEN m.is_frozen THEN 'مجمد'

        WHEN m.subscription_end_date <= CURRENT_DATE THEN 'منتهي'

        WHEN m.subscription_end_date <=
            CURRENT_DATE + (
                SELECT expiring_soon_days
                FROM settings
                LIMIT 1
            )
        THEN 'قرب ينتهي'

        ELSE 'نشط'
    END AS status_text,

    -- اللون
    CASE
        WHEN m.is_frozen THEN 'blue'

        WHEN m.subscription_end_date <= CURRENT_DATE THEN 'red'

        WHEN m.subscription_end_date <=
            CURRENT_DATE + (
                SELECT expiring_soon_days
                FROM settings
                LIMIT 1
            )
        THEN 'yellow'

        ELSE 'green'
    END AS status_color

FROM members m;

CREATE FUNCTION get_transaction_stats(period_days text)
RETURNS TABLE (
    income real,
    expense real,
    profit real,
    transactions_count bigint
)
LANGUAGE plpgsql
AS $$
DECLARE
    from_date timestamp;
BEGIN

    IF period_days = 'all' THEN

        RETURN QUERY
        SELECT
            COALESCE(SUM(CASE WHEN direction = 'income' THEN amount_paid ELSE 0 END), 0)::real,
            COALESCE(SUM(CASE WHEN direction = 'expense' THEN amount_paid ELSE 0 END), 0)::real,
            (
                COALESCE(SUM(CASE WHEN direction = 'income' THEN amount_paid ELSE 0 END), 0)
                -
                COALESCE(SUM(CASE WHEN direction = 'expense' THEN amount_paid ELSE 0 END), 0)
            )::real,
            COUNT(*)
        FROM transactions;

    ELSE

        from_date := CURRENT_DATE - (period_days::int - 1);

        RETURN QUERY
        SELECT
            COALESCE(SUM(CASE WHEN direction = 'income' THEN amount_paid ELSE 0 END), 0)::real,
            COALESCE(SUM(CASE WHEN direction = 'expense' THEN amount_paid ELSE 0 END), 0)::real,
            (
                COALESCE(SUM(CASE WHEN direction = 'income' THEN amount_paid ELSE 0 END), 0)
                -
                COALESCE(SUM(CASE WHEN direction = 'expense' THEN amount_paid ELSE 0 END), 0)
            )::real,
            COUNT(*)
        FROM transactions
        WHERE paid_at >= from_date;

    END IF;

END;
$$;
